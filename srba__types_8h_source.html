00312         size_t findRowSpan(size_t *row_min_idx=NULL, size_t *row_max_idx=NULL) const
00313         {
00314             size_t row_min=std::numeric_limits<size_t>::max();
00315             size_t row_max=0;
00316             const size_t nCols = base_t::getColCount();
00317             for (size_t j=0;j<nCols;j++)
00318                 for (typename base_t::col_t::const_iterator itRow=base_t::getCol(j).begin();itRow!=base_t::getCol(j).end();++itRow) {
00319                     mrpt::utils::keep_max(row_max, itRow->first);
00320                     mrpt::utils::keep_min(row_min, itRow->first);
00321                 }
00322             if (row_min==std::numeric_limits<size_t>::max())
00323                 row_min=0;
00324             if (row_min_idx) *row_min_idx = row_min;
00325             if (row_max_idx) *row_max_idx = row_max;
00326             return (row_max-row_min)+1;
00327         }
00328 
00330         static size_t findRowSpan(const std::vector<typename base_t::col_t*> &lstColumns, size_t *row_min_idx=NULL, size_t *row_max_idx=NULL)
00331         {
00332             size_t row_min=std::numeric_limits<size_t>::max();
00333             size_t row_max=0;
00334             for (size_t i=0;i<lstColumns.size();++i)
00335                 for (typename base_t::col_t::const_iterator itRow=lstColumns[i]->begin();itRow!=lstColumns[i]->end();++itRow) {
00336                     mrpt::utils::keep_max(row_max, itRow->first);
00337                     mrpt::utils::keep_min(row_min, itRow->first);
00338                 }
00339             if (row_min==std::numeric_limits<size_t>::max())
00340                 row_min=0;
00341             if (row_min_idx) *row_min_idx = row_min;
00342             if (row_max_idx) *row_max_idx = row_max;
00343             return (row_max-row_min)+1;
00344         }
00345 
00350         void getSparsityStats(size_t &nMaxBlocks, size_t &nNonZeroBlocks  ) const {
00351             const size_t nCols = base_t::getColCount();
00352             const size_t nRows = findRowSpan();
00353             nMaxBlocks = nCols * nRows;
00354             nNonZeroBlocks = 0;
00355             for (size_t j=0;j<nCols;j++)
00356                 nNonZeroBlocks+= base_t::getCol(j).size();
00357         }
00358 
00360         static void getSparsityStats(const std::vector<typename base_t::col_t*> &lstColumns, size_t &nMaxBlocks, size_t &nNonZeroBlocks  )
00361         {
00362             const size_t nCols = lstColumns.size();
00363             const size_t nRows = findRowSpan(lstColumns);
00364             nMaxBlocks = nCols * nRows;
00365             nNonZeroBlocks = 0;
00366             for (size_t j=0;j<nCols;j++)
00367                 nNonZeroBlocks+=lstColumns[j]->size();
00368         }
